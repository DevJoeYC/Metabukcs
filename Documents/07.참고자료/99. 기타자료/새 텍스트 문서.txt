create view pro_list
AS
	SELECT
    	pro.pro_no,
    	di.div_name		AS	name,
        de.def_hname	AS	hname,
        de.def_ename	AS	ename,
        c.cat_name		AS	cat,
        de.def_info		AS	info,
        di.volume,
        di.unit,
        pro.pro_price	AS	price,
        pro.pro_acc		AS	acc,
        pro.pro_recom	AS	flag_recom,
        pro.pro_sell	AS	flag_sell,
        pro.pro_chrcp	as	flag_rec
    FROM `st_product` pro
        LEFT JOIN div_product di
            ON pro.div_no = di.div_no 
        LEFT JOIN def_product de
            ON di.def_no = de.def_no
		LEFT JOIN category c
            ON de.cat_no = c.cat_no;
